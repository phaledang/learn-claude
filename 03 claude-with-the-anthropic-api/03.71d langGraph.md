# LangGraph agent workflow (Claude + local vector store)

> **What this sample demonstrates**
>
> * `retrieve` node → get documents from the retriever
> * `rewrite_query` node → improve retrieval when nothing useful is found
> * `generate_answer` node → answer only from retrieved context
> * conditional edges → retry retrieval once before stopping
>
> The code below is an **illustrative implementation** based on the workflow concepts described in the sources above.

```python
# pip install langgraph langchain langchain-community langchain-text-splitters \
#             langchain-anthropic langchain-openai faiss-cpu

import os
from typing import List, TypedDict

from langchain_core.documents import Document
from langchain_core.prompts import ChatPromptTemplate
from langchain_community.document_loaders import TextLoader
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_community.vectorstores import FAISS
from langchain_openai import OpenAIEmbeddings
from langchain_anthropic import ChatAnthropic

from langgraph.graph import StateGraph, START, END


# -----------------------------------------------------------------------------
# 1) Build the retrieval index
# -----------------------------------------------------------------------------
# Example corpus
loader = TextLoader("report.txt", encoding="utf-8")
docs = loader.load()

splitter = RecursiveCharacterTextSplitter(
    chunk_size=800,
    chunk_overlap=120
)
chunks = splitter.split_documents(docs)

# Embeddings + vector store
embeddings = OpenAIEmbeddings()  # can be replaced by Azure OpenAI embeddings
vectorstore = FAISS.from_documents(chunks, embeddings)
retriever = vectorstore.as_retriever(search_kwargs={"k": 4})

# Claude model
llm = ChatAnthropic(
    model="claude-3-5-sonnet-20240620",
    temperature=0
)


# -----------------------------------------------------------------------------
# 2) Define workflow state
# -----------------------------------------------------------------------------
class RAGState(TypedDict):
    question: str
    rewritten_question: str
    documents: List[Document]
    answer: str
    attempt: int


# -----------------------------------------------------------------------------
# 3) Define nodes
# -----------------------------------------------------------------------------
def retrieve(state: RAGState) -> RAGState:
    """Retrieve documents using the current question."""
    current_question = state["rewritten_question"] or state["question"]
    retrieved_docs = retriever.invoke(current_question)

    return {
        **state,
        "documents": retrieved_docs,
        "attempt": state["attempt"] + 1
    }


def rewrite_query(state: RAGState) -> RAGState:
    """Rewrite the question to improve retrieval."""
    prompt = ChatPromptTemplate.from_template(
        """Rewrite the question to improve document retrieval.
Focus on key business/domain terms and remove ambiguity.

Original question:
{question}
"""
    )

    rewritten = llm.invoke(
        prompt.format_messages(question=state["question"])
    ).content

    return {
        **state,
        "rewritten_question": rewritten
    }


def generate_answer(state: RAGState) -> RAGState:
    """Generate grounded answer from retrieved documents."""
    context = "\n\n".join(doc.page_content for doc in state["documents"])

    prompt = ChatPromptTemplate.from_template(
        """Answer the question using only the provided context.
If the context is insufficient, say: "Insufficient context."

Question:
{question}

Context:
{context}
"""
    )

    answer = llm.invoke(
        prompt.format_messages(
            question=state["question"],
            context=context
        )
    ).content

    return {
        **state,
        "answer": answer
    }


def no_answer(state: RAGState) -> RAGState:
    """Fallback when retrieval still doesn't find useful context."""
    return {
        **state,
        "answer": "Insufficient context."
    }


# -----------------------------------------------------------------------------
# 4) Routing logic
# -----------------------------------------------------------------------------
def route_after_retrieve(state: RAGState) -> str:
    """
    If docs were found, generate.
    If no docs and only one attempt so far, rewrite query and retry.
    Otherwise stop with no_answer.
    """
    if state["documents"]:
        return "generate_answer"

    if state["attempt"] < 2:
        return "rewrite_query"

    return "no_answer"


# -----------------------------------------------------------------------------
# 5) Build the graph
# -----------------------------------------------------------------------------
graph = StateGraph(RAGState)

graph.add_node("retrieve", retrieve)
graph.add_node("rewrite_query", rewrite_query)
graph.add_node("generate_answer", generate_answer)
graph.add_node("no_answer", no_answer)

graph.add_edge(START, "retrieve")

graph.add_conditional_edges(
    "retrieve",
    route_after_retrieve,
    {
        "generate_answer": "generate_answer",
        "rewrite_query": "rewrite_query",
        "no_answer": "no_answer"
    }
)

graph.add_edge("rewrite_query", "retrieve")
graph.add_edge("generate_answer", END)
graph.add_edge("no_answer", END)

app = graph.compile()


# -----------------------------------------------------------------------------
# 6) Run it
# -----------------------------------------------------------------------------
result = app.invoke(
    {
        "question": "What are the key risk factors?",
        "rewritten_question": "",
        "documents": [],
        "answer": "",
        "attempt": 0
    }
)

print("FINAL ANSWER:\n", result["answer"])
print("\nRETRIEVED CHUNKS:", len(result["documents"]))
if result["rewritten_question"]:
    print("\nREWRITTEN QUERY:\n", result["rewritten_question"])
```

***

# Why this is a LangGraph-style workflow

This example uses the capabilities typically associated with **LangGraph** rather than a plain linear chain:

* **Stateful execution**: the graph keeps the current question, rewritten question, retrieved docs, answer, and retry count in shared state. [\[futurestackdev.com\]](https://futurestackdev.com/blog/ai-engineering/langchain-langgraph-rag-guide), [\[docs.langchain.com\]](https://docs.langchain.com/oss/javascript/langchain/rag)
* **Conditional routing**: after retrieval, execution branches either to answer generation, query rewriting, or fallback logic. [\[docs.langchain.com\]](https://docs.langchain.com/oss/javascript/langchain/rag)
* **Cycle / retry loop**: the rewrite node routes back into retrieval, which is the kind of looped workflow LangGraph is designed for. [\[docs.langchain.com\]](https://docs.langchain.com/oss/javascript/langchain/rag)

# Enterprise extension: Azure AI Search as the retrieval node

For an Azure-first design, the **graph stays the same** and only the retrieval node changes. Azure AI Search is positioned as the retrieval/indexing layer for RAG and supports keyword, vector, and hybrid retrieval. [\[docs.azure.cn\]](https://docs.azure.cn/en-us/search/tutorial-rag-build-solution), [\[github.com\]](https://github.com/microsoft/aitour-rag-with-ai-search)

### Replace the `retrieve()` node with this version

```python
from azure.search.documents import SearchClient
from azure.core.credentials import AzureKeyCredential
from langchain_core.documents import Document

search_client = SearchClient(
    endpoint="https://<your-search-service>.search.windows.net",
    index_name="documents",
    credential=AzureKeyCredential(os.environ["AZURE_AI_SEARCH_KEY"])
)

def retrieve(state: RAGState) -> RAGState:
    current_question = state["rewritten_question"] or state["question"]

    # Basic keyword search example.
    # In a real system, this can be replaced with hybrid/vector queries.
    results = search_client.search(
        search_text=current_question,
        top=4
    )

    retrieved_docs = []
    for r in results:
        retrieved_docs.append(
            Document(
                page_content=r["content"],
                metadata={
                    "source": r.get("source"),
                    "id": r.get("id")
                }
            )
        )

    return {
        **state,
        "documents": retrieved_docs,
        "attempt": state["attempt"] + 1
    }
```

# When this pattern is preferable

A plain LangChain RAG chain is usually enough for **simple retrieval → answer** flows. LangGraph becomes more useful when the workflow needs **retry logic, branching, state, or multi-step control**, which is exactly how the sources describe the distinction between the two frameworks. [\[docs.langchain.com\]](https://docs.langchain.com/oss/javascript/langchain/rag), [\[blockchain...ouncil.org\]](https://www.blockchain-council.org/ai/langchain-2026-reliable-agents-langchain-rag/)
